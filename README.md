## Example

```javascript
    const { Classeviva } = require('classeviva.js');

    const classeviva = new Classeviva('USERNAME / EMAIL', 'PASSWORD');

    (async () => {
        await classeviva.login();

        classeviva.getAbsences()
        .then(absences => {
            console.log(`You were absent in school ${absences.length} times`);
        });

        setTimeout(() => {
            classeviva.logout();
            process.exit();
        }, 3500);
    })();
```
